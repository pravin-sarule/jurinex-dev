                                        <>
                                          <StreamingMarkdown key={`stream-${selectedMessageId}-${streamResetKey}`} bufferRef={streamBufferRef} scrollTargetRef={chatThreadRef} />

                                          {streamingStatus === 'generating' && !streamBufferRef.current && (
                                            <TypingDots />
                                          )}
                                        </>
