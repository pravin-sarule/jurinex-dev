// ─── Stable-split finder ─────────────────────────────────────────────────────
// Returns the char index at which the buffer can be split into a stable
// (already-complete) prefix and a live (still-growing) suffix.
//
// Strategy (in priority order):
//   1. Last \n\n outside a fenced code block — clean paragraph boundary.
//   2. Fallback for large buffers: last \n that leaves ≤ LIVE_CAP chars on the
//      right side.  This prevents the live-div parse from scaling with total
//      response length when the LLM uses single newlines (e.g. legal text,
//      bullet lists without blank-line separators).  Without this fallback,
//      parseMarkdown(entireBuffer) runs every tick → O(n) freeze.
const LIVE_CAP = 1200; // chars — max live-div parse size per tick

function findStableSplit(text) {
  // ── 1. Try double-newline (best boundary) ────────────────────────────────
  let idx = text.lastIndexOf('\n\n');
  if (idx >= 0) {
    const before = text.substring(0, idx);
    const fences = (before.match(/^```/gm) || []).length;
    if (fences % 2 === 0) return idx;
    const openFence = before.lastIndexOf('\n```');
    const fallbackIdx = openFence > 0 ? before.lastIndexOf('\n\n', openFence) : -1;
    if (fallbackIdx >= 0) return fallbackIdx;
    // If even the fallback is inside a fence, drop through to single-\n below.
  }

  // ── 2. Single-newline fallback for large buffers ─────────────────────────
  // Only engage when the buffer is bigger than LIVE_CAP so short responses
  // (first few tokens) still wait for a proper \n\n boundary.
  if (text.length > LIVE_CAP) {
    const cutoff = text.length - LIVE_CAP;
    const nlIdx  = text.lastIndexOf('\n', cutoff);
    if (nlIdx > 0) return nlIdx;
    // No newline at all — split at the cutoff position (mid-word is ugly but
    // prevents unbounded O(n) parses; only happens for wall-of-text responses).
    return cutoff;
  }

  return -1;
}
