  // ── Repetition detection constants ──────────────────────────────────────────
  // LOOP_MIN_BUFFER: ignore short responses — no false positives on brief answers.
  // LOOP_WINDOW: size of the "tail fingerprint" we look for earlier in the buffer.
  // LOOP_GAP: minimum chars between the fingerprint occurrence and the tail, so we
  //   don't fire on perfectly normal adjacent duplicate phrases.
  // LOOP_CHECK_CHARS: only run the O(n) scan after this many new chars arrive, to
  //   avoid burning CPU on every tiny chunk from a fast stream.
  const LOOP_MIN_BUFFER  = 2000;
  const LOOP_WINDOW      = 350;
  const LOOP_GAP         = 600;
  const LOOP_CHECK_CHARS = 800;

  const detectRepetitionLoop = (buffer) => {
    if (buffer.length < LOOP_MIN_BUFFER + LOOP_WINDOW + LOOP_GAP) return false;
    // The "tail fingerprint": the last LOOP_WINDOW chars, left-trimmed so leading
    // whitespace from a new paragraph doesn't make every check unique.
    const tail = buffer.slice(-LOOP_WINDOW).trimStart();
    // Require at least 80 non-whitespace chars so we don't match on padding alone.
    if (tail.replace(/\s+/g, '').length < 80) return false;
    // Only search the part of the buffer that is at least LOOP_GAP before the tail.
    const searchEnd = buffer.length - LOOP_WINDOW - LOOP_GAP;
    if (searchEnd < LOOP_MIN_BUFFER) return false;
    return buffer.slice(0, searchEnd).includes(tail);
  };

  const appendStreamChunk = (text) => {
    if (typeof text !== 'string' || !text) return;
    // Drop chunks arriving after a loop-abort so they don't pollute the display.
    if (loopAbortedRef.current) return;

    // Only update the ref — StreamingPre reads from it directly via its own interval.
    // Zero React state updates = zero parent re-renders during streaming.
    streamBufferRef.current += text;
    loopCharsRef.current   += text.length;

    // Periodic loop check: only runs every LOOP_CHECK_CHARS chars to keep overhead low.
    if (loopCharsRef.current >= LOOP_CHECK_CHARS) {
      loopCharsRef.current = 0;
      if (detectRepetitionLoop(streamBufferRef.current)) {
        loopAbortedRef.current = true;
        // Hard-close the HTTP connection so the backend stops streaming immediately.
        loopAbortControllerRef.current?.abort();
        // Append a visible notice so the user knows why the response ended early.
        streamBufferRef.current +=
          '\n\n---\n*\\[Response generation stopped: repetition loop detected. ' +
          'The model was cycling through the same content. ' +
          'Please try rephrasing your question.\\]*';
        flushStreamUiUpdate();
        console.warn('[CircuitBreaker] Repetition loop detected — stream aborted.');
      }
    }
  };
